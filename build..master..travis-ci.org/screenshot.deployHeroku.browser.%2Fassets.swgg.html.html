/*
 * initialize swgg-client
 * 1. if using browser, then embed the client in webpage:
 *   <script src="https://kaizhu256.github.io/node-utility2/build..beta..travis-ci.org/app/assets.utility2.rollup.js"></script>
 *   <script>window.utility2.corsForwardProxyHostifNeeded=function(){return "https://h1-proxy1.herokuapp.com";};</script>
 * 2. if using node, then download the client from cdn:
 *   curl -LO https://kaizhu256.github.io/node-utility2/build..beta..travis-ci.org/app/assets.utility2.rollup.js
 * 3. run the code below in browser or node to initialize swgg-client
 */
var swgg;
swgg = typeof window === "object" && window && window.swgg; // browser
swgg = swgg || require("./assets.utility2.rollup.js"); // node
swgg.apiUpdate({
  modeAjax: true,
  url: "https://h1-swgg-facebook-master.herokuapp.com/assets.swgg.swagger.json"
}, function (error) {
  if (error) {
    console.error(error);
    return;
  }
  console.log("initialized swgg-client");
});
